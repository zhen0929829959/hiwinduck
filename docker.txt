# 基底映像檔
FROM osrf/ros:humble-desktop-full

# Dockerfile 的建置參數
ARG USER=initial
ARG GROUP=initial
ARG UID=1000
ARG GID="${UID}"
ARG SHELL=/bin/bash
ARG HARDWARE=x86_64
ARG ENTRYPOINT_FILE=entrypoint.sh

# 設定NVIDIA Docker的環境變數
ENV NVIDIA_VISIBLE_DEVICES all
ENV NVIDIA_DRIVER_CAPABILITIES all

# 建立Docker使用者
RUN groupadd --gid "${GID}" "${GROUP}" \
    && useradd --gid "${GID}" --uid "${UID}" -ms "${SHELL}" "${USER}" \
    && mkdir -p /etc/sudoers.d \
    && echo "${USER}:x:${UID}:${UID}:${USER},,,:/home/${USER}:${SHELL}" >> /etc/passwd \
    && echo "${USER}:x:${UID}:" >> /etc/group \
    && echo "${USER} ALL=(ALL) NOPASSWD: ALL" > "/etc/sudoers.d/${USER}" \
    && chmod 0440 "/etc/sudoers.d/${USER}"

#APT 套件來源設定
RUN sed -i 's@archive.ubuntu.com@ftp.tku.edu.tw@g' /etc/apt/sources.list

# Time
ENV TZ=Asia/Taipei
RUN ln -snf /usr/share/zoneinfo/"${TZ}" /etc/localtime && echo "${TZ}" > /etc/timezone

# 複製專案設定檔
COPY --chmod=0775 ./${ENTRYPOINT_FILE} /entrypoint.sh
COPY --chown="${USER}":"${GROUP}" --chmod=0775 config config


# 安裝系統工具 Install packages
RUN apt update \
    && apt install -y --no-install-recommends \
        sudo \
        git \
        htop \
        wget \
        curl \
        psmisc \
        tmux \
        terminator \
        udev \
        libtool \
        python3-pip \
        python3-dev \
        python3-setuptools \
        python3-colcon-common-extensions \
        software-properties-common \
        lsb-release \
        ros-humble-rmw-cyclonedds-cpp \
    && apt clean \
    && rm -rf /var/lib/apt/lists/*

# 執行 Python 套件安裝腳本
RUN ./config/pip/pip_setup.sh
RUN export ROS_APT_SOURCE_VERSION=$(curl -s https://api.github.com/repos/ros-infrastructure/ros-apt-source/releases/latest | grep -F "tag_name" | awk -F\" '{print $4}') && \
    curl -L -o /tmp/ros2-apt-source.deb "https://github.com/ros-infrastructure/ros-apt-source/releases/download/${ROS_APT_SOURCE_VERSION}/ros2-apt-source_${ROS_APT_SOURCE_VERSION}.$(lsb_release -c | awk '{print $2}')_all.deb" && \
    dpkg -i /tmp/ros2-apt-source.deb && \
    rm /tmp/ros2-apt-source.deb

# 安裝 OMPL 與運動規劃相關依賴
RUN apt update && apt install -y --no-install-recommends \
    lsb-release \
    g++ \
    cmake \
    pkg-config \
    libboost-serialization-dev \
    libboost-filesystem-dev \
    libboost-system-dev \
    libboost-program-options-dev \
    libboost-test-dev \
    libeigen3-dev \
    libode-dev wget \
    libyaml-cpp-dev \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists 

RUN sudo pip3 install -vU https://github.com/CastXML/pygccxml/archive/develop.zip pyplusplus

RUN apt update && apt install -y --no-install-recommends \
    castxml \
    libboost-python-dev \
    libboost-numpy-dev \
    python3-numpy \
    pypy3 \
    python3-pyqt5.qtopengl \
    freeglut3-dev \
    libassimp-dev \
    python3-opengl \
    python3-flask \
    python3-celery \
    libccd-dev \
    libfcl-dev \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists 

RUN sudo pip3 install -vU PyOpenGL-accelerate


# 安裝 Intel RealSense SDK
RUN git clone https://github.com/realsenseai/librealsense.git && cd librealsense \
    && mkdir build && cd build && cmake .. && make uninstall && make clean && make && sudo make install

# 設定使用者開發環境
USER ${USER}

RUN ./config/shell/bash_setup.sh "${USER}" "${GROUP}" \
    && ./config/shell/terminator/terminator_setup.sh "${USER}" "${GROUP}" \
    && ./config/shell/tmux/tmux_setup.sh "${USER}" "${GROUP}" \
    && sudo rm -rf /config
RUN export CXX=g++
RUN export MAKEFLAGS="-j nproc"
RUN echo "source /opt/ros/humble/setup.bash" >> ~/.bashrc
RUN echo "export RMW_IMPLEMENTATION=rmw_cyclonedds_cpp" >> ~/.bashrc

# 設定 Docker 工作空間
RUN sudo mkdir -p /home/"${USER}"/work
WORKDIR /home/"${USER}"/work

# 安裝 ROS2 套件
RUN sudo apt update && sudo apt install -y --no-install-recommends \
    ros-humble-geometric-shapes \
    ros-humble-srdfdom \
    ros-humble-visp \
    python3-rosdep \
    libmodbus-dev \
    ros-humble-realsense2-camera \
    ros-humble-realsense2-camera-msgs \
    && sudo apt clean \
    && sudo rm -rf /var/lib/apt/lists/*
RUN sudo rosdep init || true
RUN rosdep update

# 安裝 Python 套件
RUN python3 -m pip install --upgrade pip
RUN sudo pip3 install \
    numpy==1.26.4 \
    scipy \
    transforms3d \
    pyserial \
    opencv-python==4.11.0.86 \
    opencv-contrib-python==4.11.0.86 \
    pupil-apriltags

#安裝 PyTorch
RUN sudo pip3 install --ignore-installed torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu124

# 安裝 YOLO
RUN python3 -m pip install \
    ultralytics \
    torch \
    torchvision \
    torchaudio

# 工作目錄
WORKDIR /home/"${USER}"/work

# 開放 SSH Port
EXPOSE 22

# Container 啟動方式
ENTRYPOINT [ "/entrypoint.sh", "terminator" ]
